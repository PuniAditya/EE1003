\begin{frame}{The Question}
The function $e^{x}$ over the interval [0,1] is to be evaluated using the Taylor series 1+x+$\frac{x^{2}}{2!}$+$\frac{x^{3}}{3!}$+... to accuracy of $\delta>0$.The number of terms in the series that is considered for this accuracy is $n$. Then
\begin{enumerate}
	\item for a given $x \in [0,1]$ and a given $\delta$, there is no finite n that is valid
    \item for a given $\delta > 0$, there is a valid n that is finite for a given $x \in[0,1] $, but there is no finite n that is valid for all $x \in[0,1]$
    \item for a given $\delta >0$,there is a finite n that is valid for all $x \in [0,1]$
    \item there is a finite n that is valid for all x in [0,1] an all $\delta >0$
\end{enumerate}
\end{frame}

\begin{frame}{Lagrange Remainder}
For a function $f\brak{x}$ expanded around $a=0$ (Maclaurin series), the exact error $R_n\brak{x}$ after $n$ terms is given by Lagrange's form:
\begin{align}
    R_n\brak{x} = \frac{f^{\brak{n}}\brak{c}}{n!} x^n
\end{align}
where $c$ is some real number strictly between $0$ and $x$.

For the specific function $f\brak{x} = e^x$:
\begin{align}
    f^{\brak{n}}\brak{x} &= e^x \quad \forall n \ge 1 \\
    \implies f^{\brak{n}}\brak{c} &= e^c
\end{align}

Substituting $f^{\brak{n}}\brak{c}$ into the remainder formula yields the error term for our series:
\begin{align}
    R_n\brak{x} &= \frac{e^c}{n!} x^n
\end{align}
\end{frame}

\begin{frame}{Bounding the Error ($\epsilon$)}
To guarantee accuracy $\forall x \in \sbrak{0, 1}$, we evaluate the worst-case error scenario:
\begin{align}
    x \in \sbrak{0,1} &\implies x^n \le 1 \\
    c \in \sbrak{0,x} &\implies e^c \le e^1 = e
\end{align}

Applying these maximums to the Lagrange remainder gives the absolute upper bound for the actual error $\epsilon$:
\begin{align}
    \epsilon &= \abs{R_n\brak{x}} \le \frac{e}{n!}
\end{align}
Note: The upper bound $\frac{e}{n!}$ is constant and completely independent of $x$.
\end{frame}

\begin{frame}{Accuracy ($\delta$)}
Target requirement: The actual error must be within the given tolerance $\delta$.
\begin{align}
    \epsilon \le \delta
\end{align}

By forcing worst-case bound to be less than $\delta$, we guarantee the requirement is met for all $x$:
\begin{align}
    \epsilon \le \frac{e}{n!} \le \delta
\end{align}

Solving for the required number of terms $n$:
\begin{align}
    \frac{e}{n!} \le \delta \implies n! \ge \frac{e}{\delta}
\end{align}

\vfill
\textbf{Conclusion:}\\
Since $\lim_{n \to \infty} n! = \infty$, $\exists$ a finite $n$ satisfying $n! \ge \frac{e}{\delta}$ for any given $\delta > 0$. Because this $n$ was derived from the worst-case bound, it is valid $\forall x \in \sbrak{0, 1}$. Option (3) is correct.
\end{frame}

\begin{frame}[fragile]{Python Implementation}
\begin{lstlisting}[language=Python]
import numpy as np
import matplotlib.pyplot as plt
from scipy.special import factorial

x = np.linspace(0, 1, 500)
n = 3

ex = np.exp(x)
sn = sum((x**k) / factorial(k) for k in range(n))
error = np.abs(ex - sn)

delta_assumed = np.exp(1) / factorial(n)
delta_measured = np.max(error)

plt.figure(figsize=(8, 6))
plt.plot(x, ex, label=r'$e^x$')
plt.plot(x, sn, label=r'$S_n(x)$')
plt.plot(x, error, label=r'$\epsilon = \vert e^x - S_n(x)\vert$', color='red')

plt.axhline(delta_assumed, color='purple', linestyle='--', label=r'Assumed $\delta = \frac{e}{n!}$')
plt.axhline(delta_measured, color='orange', linestyle=':', label=r'Measured $\delta = \max(\epsilon)$')
plt.axhline(0, color='black', linewidth=2)
plt.axvline(0, color='black', linewidth=2)
plt.xlabel(r'$x$')
plt.ylabel(r'$y$')
plt.grid(True, linestyle=':', alpha=0.7)
plt.legend()
plt.tight_layout()
plt.savefig('../figs/plot.jpg')
plt.show()
\end{lstlisting}
