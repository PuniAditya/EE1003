\begin{frame}{Question}
$P_X(x) = M\exp(-2|x|) + N\exp(-3|x|)$ is the probability density function for the real random variable $X$, over the entire $x$ axis. $M$ and $N$ are both positive real numbers. The equation relating $M$ and $N$ is
\begin{enumerate}
    \item $M + \frac{2}{3} N = 1$
    \item $2M + \frac{1}{3}N = 1$
    \item $M + N = 1$
    \item $M + N = 3$
\end{enumerate}
\end{frame}

\begin{frame}{Total Probability Condition}
For any valid Probability Density Function (PDF) $P_X(x)$, the total area under the curve over the entire real axis must equal exactly $1$:
\begin{align}
    \int_{-\infty}^{\infty} P_X(x) \, dx &= 1
\end{align}

Substituting the given PDF:
\begin{align}
    \int_{-\infty}^{\infty} \brak{M\exp\brak{-2\abs{x}} + N\exp\brak{-3\abs{x}}} \, dx &= 1
\end{align}
\end{frame}

\begin{frame}{Theoretical Solution}
Since $\exp\brak{-a\abs{x}}$ is an even function, we evaluate over $[0, \infty)$:
\begin{align}
    \int_{-\infty}^{\infty} \exp\brak{-a\abs{x}} \, dx &= 2 \int_{0}^{\infty} \exp\brak{-ax} \, dx \\
    &= 2 \sbrak{\frac{\exp\brak{-ax}}{-a}}_0^\infty \\
    &= 2 \brak{0 - \frac{1}{-a}} = \frac{2}{a}
\end{align}

Applying this result for $a=2$ and $a=3$:
\begin{align}
    M\brak{\frac{2}{2}} + N\brak{\frac{2}{3}} &= 1 \\
    M + \frac{2}{3}N &= 1
\end{align}

\textbf{Conclusion:} Option (1) is correct.
\end{frame}

\begin{frame}[fragile]{Python Implementation: Computation}
\begin{lstlisting}[language=Python]
import sympy as sp
import numpy as np
import matplotlib.pyplot as plt

x, t = sp.symbols('x t', real=True)
M, N = sp.symbols('M N', real=True, positive=True)
pdf_sym = sp.Piecewise(
    (M * sp.exp(2 * x) + N * sp.exp(3 * x), x < 0),
    (M * sp.exp(-2 * x) + N * sp.exp(-3 * x), True)
)
total_prob = sp.integrate(pdf_sym, (x, -sp.oo, sp.oo))
M_sol = sp.solve(total_prob - 1, M)[0]

N_val = 0.6
M_val = M_sol.subs(N, N_val)
pdf_sub = pdf_sym.subs({M: M_val, N: N_val})

cdf_sym = sp.integrate(pdf_sub.subs(x, t), (t, -sp.oo, x))
pdf_func = sp.lambdify(x, pdf_sub, modules=['numpy'])
cdf_func = sp.lambdify(x, cdf_sym, modules=['numpy'])

x_vals = np.linspace(-4, 4, 1000)
pdf_vals = pdf_func(x_vals)
cdf_vals = cdf_func(x_vals)
\end{lstlisting}
\end{frame}

\begin{frame}{Generating Laplace via Conditional Gaussian}
To generate samples for our PDF, we use the property that a Laplace distribution can be expressed as a \textbf{Gaussian Scale Mixture} (a Conditional Gaussian). 

If a random variable $X$ follows a Laplace distribution $f(x) = \frac{\lambda}{2} \exp\brak{-\lambda|x|}$, it can be generated in two steps:
\begin{enumerate}
    \item Draw a variance $V$ from an Exponential distribution with rate parameter $\theta = \frac{\lambda^2}{2}$.
    \item Draw $X$ from a Normal (Gaussian) distribution conditioned on that variance $V$, so $X|V \sim \mathcal{N}(0, V)$.
\end{enumerate}
This powerful property allows us to use standard Gaussian generators to build sharp-peaked, heavy-tailed distributions.
\end{frame}

\begin{frame}{Decomposing the Target PDF}
Our target PDF is $P_X(x) = M\exp\brak{-2\abs{x}} + N\exp\brak{-3\abs{x}}$. 
We must rewrite this as a mixture of standard Laplace PDFs:
\begin{align}
    P_X(x) &= M \brak{\frac{2}{2}\exp\brak{-2\abs{x}}} + N \brak{\frac{2}{3} \cdot \frac{3}{2}\exp\brak{-3\abs{x}}} \\
    P_X(x) &= \underbrace{M}_{\text{Weight } w_1} \underbrace{\brak{\frac{2}{2}\exp\brak{-2\abs{x}}}}_{\text{Laplace}(\lambda=2)} + \underbrace{\brak{\frac{2N}{3}}}_{\text{Weight } w_2} \underbrace{\brak{\frac{3}{2}\exp\brak{-3\abs{x}}}}_{\text{Laplace}(\lambda=3)}
\end{align}

Since $M + \frac{2N}{3} = 1$, this perfectly represents a weighted coin flip between two Laplace distributions.
\end{frame}

\begin{frame}{Algorithm for Sample Generation}
Based on the decomposition, we generate samples $X$ as follows:
\begin{itemize}
    \item \textbf{Step 1 (Coin Flip):} Generate $U \sim \text{Uniform}(0, 1)$.
    \item \textbf{Step 2 (Draw Variance $V$):} 
    \begin{itemize}
        \item If $U < M$: We select the $\lambda=2$ term. Draw $V \sim \text{Exp}\brak{\text{rate} = \frac{2^2}{2} = 2}$. 
        \item If $U \ge M$: We select the $\lambda=3$ term. Draw $V \sim \text{Exp}\brak{\text{rate} = \frac{3^2}{2} = 4.5}$. 
    \end{itemize}
    \item \textbf{Step 3 (Conditional Gaussian):} Draw our final sample $X \sim \mathcal{N}(0, V)$.
\end{itemize}
\textit{Note: In Python's numpy, the exponential distribution takes the `scale` parameter, which is $\frac{1}{\text{rate}}$.}
\end{frame}

\begin{frame}[fragile]{Python Implementation: Sampling Logic}
\begin{lstlisting}[language=Python]
import numpy as np
import matplotlib.pyplot as plt

N = 0.6
M = 1 - (2/3) * N  # Equals 0.6
num_samples = 10000

u = np.random.rand(num_samples)
is_term = u < M

# For lambda = 2, rate = 2 -> scale = 1/2
# For lambda = 3, rate = 4.5 -> scale = 1/4.5
V = np.zeros(num_samples)
V[is_term] = np.random.exponential(scale=1/2, size=np.sum(is_term))
V[~is_term] = np.random.exponential(scale=1/4.5, size=np.sum(~is_term))

samples = np.random.normal(loc=0, scale=np.sqrt(V))
\end{lstlisting}
\end{frame}
